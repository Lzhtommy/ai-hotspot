```typescript
import { redirect } from 'next/navigation';
import { auth } from './index';
import type { Session } from 'next-auth';

export type AdminSession = Session & { user: { id: string; role: 'admin' } };

export async function requireAdmin(): Promise<AdminSession> {
  const session = await auth();
  if (!session?.user?.id) redirect('/');
  if ((session.user as { role?: string }).role !== 'admin') redirect('/admin/access-denied');
  return session as AdminSession;
}

export class AdminAuthError extends Error {
  constructor(public code: 'UNAUTHENTICATED' | 'FORBIDDEN') {
    super(code);
  }
}

export function assertAdmin(session: Session | null): asserts session is AdminSession {
  if (!session?.user?.id) throw new AdminAuthError('UNAUTHENTICATED');
  if ((session.user as { role?: string }).role !== 'admin') throw new AdminAuthError('FORBIDDEN');
}
```

Create `src/middleware.ts` using Auth.js v5 edge-compatible auth check. Since we use database sessions, we can only read the session cookie at edge without DB hit. Strategy: edge middleware matches `/admin/:path*` and checks for the PRESENCE of the session cookie (authjs.session-token or __Secure-authjs.session-token). If absent, redirect to `/` with `?next=/admin/...`. If present, pass through — the RSC layout (Task 3) does the full DB-backed role check.

```typescript
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl;
  if (!pathname.startsWith('/admin')) return NextResponse.next();

  // Database sessions: edge can only verify cookie presence, not validity.
  // The RSC admin layout (src/app/admin/layout.tsx) performs the DB-backed
  // role check via requireAdmin(). This middleware is a cheap first filter
  // that short-circuits obviously-anonymous traffic at the edge.
  const cookieName = req.nextUrl.protocol === 'https:'
    ? '__Secure-authjs.session-token'
    : 'authjs.session-token';
  const hasSession = req.cookies.has(cookieName);
  if (!hasSession) {
    const url = req.nextUrl.clone();
    url.pathname = '/';
    url.searchParams.set('next', pathname);
    return NextResponse.redirect(url);
  }
  return NextResponse.next();
}

export const config = {
  matcher: ['/admin/:path*'],
};
```
