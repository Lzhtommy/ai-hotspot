```sql
-- Phase 6 Plan 06-01: admin + ops schema extensions.
-- Idempotent — safe to re-run.
ALTER TABLE sources ADD COLUMN IF NOT EXISTS deleted_at TIMESTAMPTZ;
ALTER TABLE sources ADD COLUMN IF NOT EXISTS category TEXT;

ALTER TABLE users ADD COLUMN IF NOT EXISTS banned_at TIMESTAMPTZ;
ALTER TABLE users ADD COLUMN IF NOT EXISTS banned_by UUID;

-- Self-referencing FK. ON DELETE SET NULL so deleting an admin user does not
-- cascade-wipe ban records for users they banned.
DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint
    WHERE conname = 'users_banned_by_fk'
  ) THEN
    ALTER TABLE users
      ADD CONSTRAINT users_banned_by_fk
      FOREIGN KEY (banned_by) REFERENCES users(id) ON DELETE SET NULL;
  END IF;
END$$;

-- Index for soft-delete-aware queries (sources list filtering).
CREATE INDEX IF NOT EXISTS sources_deleted_at_idx ON sources (deleted_at);
```

```typescript
#!/usr/bin/env tsx
/**
 * Phase 6 Plan 06-01 — admin + ops schema migration applier.
 *
 * Reads drizzle/0005_admin_ops.sql and executes against the live Neon dev branch
 * via @neondatabase/serverless HTTP client. Non-TTY safe (mirrors the 05-01
 * apply-0004-auth.ts precedent established when drizzle-kit push was blocked
 * by interactive prompts).
 *
 * Usage: pnpm db:apply:0005
 *
 * Idempotent — uses ADD COLUMN IF NOT EXISTS + DO $$ IF NOT EXISTS constraint
 * block so re-runs are safe.
 */
import { readFileSync } from 'node:fs';
import { resolve } from 'node:path';
import { neon } from '@neondatabase/serverless';

async function main(): Promise<boolean> {
  const url = process.env.DATABASE_URL;
  if (!url) {
    console.error('DATABASE_URL missing');
    return false;
  }
  const sql = neon(url);
  const migrationPath = resolve(process.cwd(), 'drizzle/0005_admin_ops.sql');
  const raw = readFileSync(migrationPath, 'utf8');
  // Split by statement — use a conservative splitter: semicolons at end of lines
  // outside DO $$ blocks. To keep correctness, split on sentinel: a single blank
  // line separating top-level statements. If DO $$ ... $$; is the only multi-
  // statement block, execute it whole.
  // Simpler: execute as one payload. neon() HTTP accepts multiple statements.
  await sql.unsafe(raw);
  console.log('OK — 0005_admin_ops applied');
  return true;
}

main().then((ok) => process.exit(ok ? 0 : 1)).catch((e) => {
  console.error(e instanceof Error ? e.message : String(e));
  process.exit(1);
});
```
