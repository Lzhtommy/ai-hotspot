```typescript
import { and, desc, eq, sql as dsql } from 'drizzle-orm';
import { db as realDb } from '@/lib/db/client';
import { items, sources } from '@/lib/db/schema';

export interface DeadLetterRow {
  id: string; // bigint serialized to string
  title: string;
  sourceName: string | null;
  failureReason: string | null;
  retryCount: number;
  processedAt: Date | null;
  ingestedAt: Date;
  url: string;
}

export async function listDeadLetterItems(
  opts: { limit: number },
  deps: { db?: typeof realDb } = {},
): Promise<DeadLetterRow[]> {
  const d = deps.db ?? realDb;
  const rows = await d
    .select({
      id: items.id,
      title: items.title,
      sourceName: sources.name,
      failureReason: items.failureReason,
      retryCount: items.retryCount,
      processedAt: items.processedAt,
      ingestedAt: items.ingestedAt,
      url: items.url,
    })
    .from(items)
    .leftJoin(sources, eq(sources.id, items.sourceId))
    .where(eq(items.status, 'dead_letter'))
    .orderBy(desc(items.processedAt))
    .limit(opts.limit);
  return rows.map((r) => ({ ...r, id: String(r.id) }));
}

export async function retryItemCore(
  input: { itemId: bigint },
  deps: { db?: typeof realDb } = {},
): Promise<{ retried: boolean }> {
  const d = deps.db ?? realDb;
  const updated = await d
    .update(items)
    .set({ status: 'pending', failureReason: null, processedAt: null, retryCount: dsql`${items.retryCount} + 1` })
    .where(and(eq(items.id, input.itemId), eq(items.status, 'dead_letter')))
    .returning({ id: items.id });
  return { retried: updated.length > 0 };
}

export async function retryAllCore(
  input: { limit: number },
  deps: { db?: typeof realDb } = {},
): Promise<{ count: number }> {
  const d = deps.db ?? realDb;
  // Select up to `limit` dead_letter rows, then update in a single WHERE IN.
  const targets = await d
    .select({ id: items.id })
    .from(items)
    .where(eq(items.status, 'dead_letter'))
    .orderBy(desc(items.processedAt))
    .limit(input.limit);
  if (targets.length === 0) return { count: 0 };
  const ids = targets.map((t) => t.id);
  await d.execute(dsql`
    UPDATE items
    SET status = 'pending',
        failure_reason = NULL,
        processed_at = NULL,
        retry_count = retry_count + 1
    WHERE id IN ${ids} AND status = 'dead_letter'
  `);
  return { count: ids.length };
}
```

```typescript
'use server';
import { revalidatePath } from 'next/cache';
import { z } from 'zod';
import { Ratelimit } from '@upstash/ratelimit';
import { auth } from '@/lib/auth';
import { assertAdmin, AdminAuthError } from '@/lib/auth/admin';
import { redis } from '@/lib/redis/client';
import { retryItemCore, retryAllCore } from '@/lib/admin/dead-letter-repo';

const BULK_LIMIT = 20;      // max items per single bulk action

// BLOCKER-3 fix: use @upstash/ratelimit sliding-window (NOT a hand-rolled tumbling
// bucket — the previous implementation allowed 40 retries in ~2s at the minute
// boundary). Sliding window enforces a true 60-second rolling cap.
// Requires: pnpm add @upstash/ratelimit (add to Task 1 install step if not already
// a dep per CLAUDE.md supporting libraries table).
const ratelimit = new Ratelimit({
  redis,
  limiter: Ratelimit.slidingWindow(20, '60 s'),
  analytics: false,
  prefix: 'admin:retry',
});

async function checkRateLimit(adminUserId: string): Promise<boolean> {
  const { success } = await ratelimit.limit(`admin:retry:${adminUserId}`);
  return success;
}

const IdSchema = z.object({ itemId: z.string().regex(/^\d+$/) });

export async function retryItemAction(input: { itemId: string }): Promise<{ ok: true; retried: boolean } | { ok: false; error: string }> {
  try {
    const session = await auth();
    assertAdmin(session);
    const parsed = IdSchema.safeParse(input);
    if (!parsed.success) return { ok: false, error: 'VALIDATION' };
    const allowed = await checkRateLimit(session.user.id);
    if (!allowed) return { ok: false, error: 'RATE_LIMITED' };
    const res = await retryItemCore({ itemId: BigInt(parsed.data.itemId) });
    revalidatePath('/admin/dead-letter');
    return { ok: true, retried: res.retried };
  } catch (e) {
    if (e instanceof AdminAuthError) return { ok: false, error: e.code };
    return { ok: false, error: 'INTERNAL' };
  }
}

export async function retryAllAction(): Promise<{ ok: true; count: number } | { ok: false; error: string }> {
  try {
    const session = await auth();
    assertAdmin(session);
    const allowed = await checkRateLimit(session.user.id);
    if (!allowed) return { ok: false, error: 'RATE_LIMITED' };
    const res = await retryAllCore({ limit: BULK_LIMIT });
    revalidatePath('/admin/dead-letter');
    return { ok: true, count: res.count };
  } catch (e) {
    if (e instanceof AdminAuthError) return { ok: false, error: e.code };
    return { ok: false, error: 'INTERNAL' };
  }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| admin browser → retry action | Untrusted itemId → zod parse |
| retry action → items row transition | Race-guarded by WHERE status='dead_letter' |
| rate limit store | Upstash Redis (existing Phase 1 singleton) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-6-50 | Elevation of Privilege | retry actions | mitigate | assertAdmin(await auth()) first, before any DB or Redis touch. |
| T-6-51 | Denial of Service / budget burn | retry loop | mitigate | Upstash sliding-window rate limit: 20 retries per admin per 60s. Server returns RATE_LIMITED; UI surfaces message. Bulk retry capped at 20 items per click. |
| T-6-52 | Race — concurrent retry of same item | double-transition | mitigate | retryItemCore includes WHERE status='dead_letter' in UPDATE. If another admin already retried, the second UPDATE hits zero rows and returns {retried: false}. |
| T-6-53 | Tampering (IDOR) | itemId param | mitigate | zod /^\d+$/ validation + BigInt coercion. Admin role gate is phase-wide; no per-record ownership model. |
| T-6-54 | Information Disclosure | failure_reason contents | accept | May contain LLM error detail; admins are trusted audience. Truncated to 80 chars in UI to avoid accidentally pasting full stack traces. |
| T-6-55 | Stale UI after retry | inconsistent state | mitigate | revalidatePath('/admin/dead-letter') after each action — row disappears from list on refresh (status is now 'pending', poller will process or republish). |
</threat_model>
