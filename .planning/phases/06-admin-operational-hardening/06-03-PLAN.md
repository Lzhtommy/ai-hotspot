```typescript
import { eq, desc, and, sql as dsql } from 'drizzle-orm';
import { db as realDb } from '@/lib/db/client';
import { users, sessions, accounts } from '@/lib/db/schema';

export class SelfBanError extends Error { constructor() { super('SELF_BAN'); } }
export class UserNotFoundError extends Error { constructor() { super('NOT_FOUND'); } }

export interface UserAdminRow {
  id: string;
  email: string;
  name: string | null;
  role: 'user' | 'admin';
  isBanned: boolean;
  bannedAt: Date | null;
  bannedBy: string | null;
  createdAt: Date;
  providers: string[]; // derived from accounts; may be empty for magic-link-only users
}

export async function listUsersForAdmin(deps: { db?: typeof realDb } = {}): Promise<UserAdminRow[]> {
  const d = deps.db ?? realDb;
  // Group-concat providers via SQL aggregation (neon-pg supports string_agg).
  const rows = await d.execute(dsql`
    SELECT u.id, u.email, u.name, u.role, u.is_banned, u.banned_at, u.banned_by, u.created_at,
           COALESCE(array_agg(a.provider) FILTER (WHERE a.provider IS NOT NULL), '{}') AS providers
    FROM users u
    LEFT JOIN accounts a ON a."userId" = u.id
    GROUP BY u.id
    ORDER BY u.created_at DESC
  `);
  return (rows as unknown as { rows: Array<Record<string, unknown>> }).rows.map((r) => ({
    id: r.id as string,
    email: r.email as string,
    name: r.name as string | null,
    role: r.role as 'user' | 'admin',
    isBanned: r.is_banned as boolean,
    bannedAt: r.banned_at ? new Date(r.banned_at as string) : null,
    bannedBy: r.banned_by as string | null,
    createdAt: new Date(r.created_at as string),
    providers: (r.providers as string[]) ?? [],
  }));
}

export async function banUserCore(
  input: { targetUserId: string; adminUserId: string },
  deps: { db?: typeof realDb } = {},
): Promise<void> {
  if (input.targetUserId === input.adminUserId) throw new SelfBanError();
  const d = deps.db ?? realDb;
  await d.transaction(async (tx) => {
    const updated = await tx
      .update(users)
      .set({ isBanned: true, bannedAt: new Date(), bannedBy: input.adminUserId })
      .where(eq(users.id, input.targetUserId))
      .returning({ id: users.id });
    if (updated.length === 0) throw new UserNotFoundError();
    // Session revocation — REQUIRED by Phase 6 SC#3: "ban revokes their session".
    // Deleting the row invalidates every outstanding cookie for this user.
    await tx.delete(sessions).where(eq(sessions.userId, input.targetUserId));
  });
}

export async function unbanUserCore(input: { targetUserId: string }, deps: { db?: typeof realDb } = {}): Promise<void> {
  const d = deps.db ?? realDb;
  await d.update(users).set({ isBanned: false, bannedAt: null, bannedBy: null }).where(eq(users.id, input.targetUserId));
}
```

import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { makeTestDb } from '../helpers/test-db';
import { banUserCore, unbanUserCore } from '@/lib/admin/users-repo';
import { users, sessions } from '@/lib/db/schema';
import { eq } from 'drizzle-orm';

describe('banUserCore — atomic session revocation', () => {
  const db = makeTestDb();

  let adminId: string;
  let targetId: string;

  beforeEach(async () => {
    // Insert admin + target
    const [admin] = await db.insert(users).values({ email: `admin-${Date.now()}@test`, role: 'admin' }).returning();
    const [target] = await db.insert(users).values({ email: `target-${Date.now()}@test`, role: 'user' }).returning();
    adminId = admin.id;
    targetId = target.id;
    // Insert 2 sessions for target
    await db.insert(sessions).values([
      { sessionToken: `tok-${Date.now()}-1`, userId: targetId, expires: new Date(Date.now() + 86400000) },
      { sessionToken: `tok-${Date.now()}-2`, userId: targetId, expires: new Date(Date.now() + 86400000) },
    ]);
  });

  afterEach(async () => {
    await db.delete(sessions).where(eq(sessions.userId, targetId));
    await db.delete(users).where(eq(users.id, targetId));
    await db.delete(users).where(eq(users.id, adminId));
  });

  it('ban deletes all sessions rows and sets audit columns atomically', async () => {
    await banUserCore({ targetUserId: targetId, adminUserId: adminId }, { db });
    const [after] = await db.select().from(users).where(eq(users.id, targetId));
    expect(after.isBanned).toBe(true);
    expect(after.bannedBy).toBe(adminId);
    expect(after.bannedAt).toBeInstanceOf(Date);
    const remainingSessions = await db.select().from(sessions).where(eq(sessions.userId, targetId));
    expect(remainingSessions).toHaveLength(0);
  });

  it('unban clears ban columns but does not restore sessions', async () => {
    await banUserCore({ targetUserId: targetId, adminUserId: adminId }, { db });
    await unbanUserCore({ targetUserId: targetId }, { db });
    const [after] = await db.select().from(users).where(eq(users.id, targetId));
    expect(after.isBanned).toBe(false);
    expect(after.bannedAt).toBeNull();
    expect(after.bannedBy).toBeNull();
    const remainingSessions = await db.select().from(sessions).where(eq(sessions.userId, targetId));
    expect(remainingSessions).toHaveLength(0);
  });
});

'use server';
import { revalidatePath } from 'next/cache';
import { z } from 'zod';
import { auth } from '@/lib/auth';
import { assertAdmin, AdminAuthError } from '@/lib/auth/admin';
import { banUserCore, unbanUserCore, SelfBanError, UserNotFoundError } from '@/lib/admin/users-repo';

const IdSchema = z.object({ targetUserId: z.string().uuid() });

export async function banUserAction(input: { targetUserId: string }): Promise<{ ok: true } | { ok: false; error: string }> {
  try {
    const session = await auth();
    assertAdmin(session);
    const parsed = IdSchema.safeParse(input);
    if (!parsed.success) return { ok: false, error: 'VALIDATION' };
    await banUserCore({ targetUserId: parsed.data.targetUserId, adminUserId: session.user.id });
    revalidatePath('/admin/users');
    return { ok: true };
  } catch (e) {
    if (e instanceof SelfBanError) return { ok: false, error: 'SELF_BAN' };
    if (e instanceof UserNotFoundError) return { ok: false, error: 'NOT_FOUND' };
    if (e instanceof AdminAuthError) return { ok: false, error: e.code };
    return { ok: false, error: 'INTERNAL' };
  }
}

export async function unbanUserAction(input: { targetUserId: string }) { /* mirror */ }
