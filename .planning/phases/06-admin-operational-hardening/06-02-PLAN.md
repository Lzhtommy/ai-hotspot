```typescript
import { and, desc, eq, isNull } from 'drizzle-orm';
import { db as realDb } from '@/lib/db/client';
import { sources } from '@/lib/db/schema';

export interface SourceAdminRow {
  id: number;
  name: string;
  rssUrl: string;
  language: string;
  weight: string;
  isActive: boolean;
  category: string | null;
  consecutiveEmptyCount: number;
  consecutiveErrorCount: number;
  lastFetchedAt: Date | null;
  createdAt: Date;
}

export async function listSourcesForAdmin(deps: { db?: typeof realDb } = {}): Promise<SourceAdminRow[]> {
  const d = deps.db ?? realDb;
  return d
    .select({ /* all columns above */ })
    .from(sources)
    .where(isNull(sources.deletedAt))
    .orderBy(desc(sources.createdAt));
}

export async function getSourceByIdForAdmin(id: number, deps: { db?: typeof realDb } = {}): Promise<SourceAdminRow | null> { ... }

export async function createSourceCore(input: {
  name: string;
  rssUrl: string;
  language?: 'zh' | 'en';
  weight?: string;
  category?: string | null;
  isActive?: boolean;
}, deps: { db?: typeof realDb } = {}): Promise<{ id: number }> { ... }

export async function updateSourceCore(id: number, patch: Partial<Pick<SourceAdminRow, 'name' | 'weight' | 'isActive' | 'category'>>, deps: { db?: typeof realDb } = {}): Promise<void> { ... }

export async function softDeleteSourceCore(id: number, deps: { db?: typeof realDb } = {}): Promise<void> {
  const d = deps.db ?? realDb;
  await d.update(sources).set({ deletedAt: new Date(), isActive: false }).where(and(eq(sources.id, id), isNull(sources.deletedAt)));
}

export function computeSourceHealth(s: Pick<SourceAdminRow, 'consecutiveEmptyCount' | 'consecutiveErrorCount'>): 'green' | 'yellow' | 'red' {
  if (s.consecutiveEmptyCount >= 3 || s.consecutiveErrorCount >= 3) return 'red';
  if (s.consecutiveEmptyCount >= 1 || s.consecutiveErrorCount >= 1) return 'yellow';
  return 'green';
}
```

```typescript
'use server';
import { revalidatePath } from 'next/cache';
import { z } from 'zod';
import { auth } from '@/lib/auth';
import { assertAdmin, AdminAuthError } from '@/lib/auth/admin';
import {
  createSourceCore,
  updateSourceCore,
  softDeleteSourceCore,
} from '@/lib/admin/sources-repo';

const SourceCreateSchema = z.object({
  name: z.string().min(1).max(200),
  rssUrl: z.string().url().max(2000),
  language: z.enum(['zh', 'en']).default('zh'),
  weight: z.string().regex(/^\d+(\.\d)?$/), // numeric(3,1)
  category: z.string().max(40).nullable().optional(),
  isActive: z.boolean().default(true),
});

const SourceUpdateSchema = z.object({
  id: z.coerce.number().int().positive(),
  name: z.string().min(1).max(200).optional(),
  weight: z.string().regex(/^\d+(\.\d)?$/).optional(),
  isActive: z.boolean().optional(),
  category: z.string().max(40).nullable().optional(),
});

export async function createSourceAction(formData: FormData): Promise<{ ok: true; id: number } | { ok: false; error: string }> {
  try {
    assertAdmin(await auth());
    const parsed = SourceCreateSchema.safeParse({ ... });
    if (!parsed.success) return { ok: false, error: 'VALIDATION' };
    const { id } = await createSourceCore(parsed.data);
    revalidatePath('/admin/sources');
    return { ok: true, id };
  } catch (e) {
    if (e instanceof AdminAuthError) return { ok: false, error: e.code };
    return { ok: false, error: 'INTERNAL' };
  }
}

// Mirror shape: updateSourceAction, softDeleteSourceAction (takes id: number), toggleActiveAction (takes id + isActive).
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| admin browser form → Server Action | Form input is untrusted until zod-parsed |
| Server Action → DB | Parameters bound via Drizzle ORM (no string concat SQL) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-6-20 | Elevation of Privilege | admin-sources Server Actions | mitigate | Every action calls assertAdmin(await auth()) FIRST — before zod parse, before DB touch. AdminAuthError surfaces as generic error code to client. |
| T-6-21 | Tampering (IDOR) | updateSourceAction / softDeleteSourceAction | mitigate | Only admins reach the action. id is zod-parsed to positive int. No per-record ownership model — admins have full access by design. |
| T-6-22 | Injection | rssUrl field | mitigate | z.string().url() rejects malformed URLs before DB insert. Drizzle binds params. Ingestion wrapper (fetchRSSHub, Phase 1) already scrubs keys from error paths. |
| T-6-23 | Denial of Service | createSourceAction spam | accept | Admin-only; admins are trusted. No rate limit needed. Phase 6 OPS plans add global Sentry monitoring so outlier behavior is visible. |
| T-6-24 | Information Disclosure | error responses | mitigate | Actions return opaque codes ('VALIDATION', 'INTERNAL'), never err.message. Prevents leaking DB schema or SDK error details. |
| T-6-25 | Data loss via hard-delete | softDeleteSourceAction | mitigate | Only sets deleted_at — items remain. Confirmed by reading fetch-source-core to ensure items FK to sources does not cascade on SET deleted_at. |
</threat_model>
