    ```tsx
    'use client';

    /**
     * UserChip — Phase 5 AUTH-06, AUTH-08, D-18.
     *
     * Three states: anonymous / authenticated-with-image / authenticated-without-image.
     * Anonymous branch unchanged from Phase 4. Authenticated branches new.
     *
     * Consumed by:
     *   - src/components/layout/sidebar.tsx (RSC passes session prop via auth())
     */

    import { useRef, useState, useEffect } from 'react';
    import Image from 'next/image';
    import { Button } from '@/components/layout/button';
    import { Icon } from '@/components/layout/icon';
    import { signOutAction } from '@/server/actions/auth';

    interface SessionUser {
      id: string;
      email: string;
      name: string | null;
      image: string | null;
      role?: string;
    }

    interface UserChipProps {
      session: { user: SessionUser } | null;
    }

    function truncateName(name: string | null, email: string): string {
      const raw = name ?? email.split('@')[0] ?? 'User';
      return raw.length > 8 ? raw.slice(0, 8) + '…' : raw;
    }

    function initialOf(name: string | null, email: string): string {
      const raw = name ?? email.split('@')[0] ?? 'U';
      return raw.charAt(0).toUpperCase();
    }

    export function UserChip({ session }: UserChipProps) {
      // Anonymous branch — PRESERVED from Phase 4, unchanged
      if (!session?.user) {
        return (
          <div style={{ margin: '0 12px 12px' }}>
            <Button
              variant="ghost"
              size="sm"
              onClick={() => document.dispatchEvent(new CustomEvent('open-login-modal'))}
            >
              登录
            </Button>
          </div>
        );
      }

      // Authenticated — with popover
      const { user } = session;
      const displayName = truncateName(user.name, user.email);
      const [open, setOpen] = useState(false);
      const containerRef = useRef<HTMLDivElement>(null);

      // click-outside closes
      useEffect(() => {
        if (!open) return;
        function handleClickOutside(e: MouseEvent) {
          if (containerRef.current && !containerRef.current.contains(e.target as Node)) {
            setOpen(false);
          }
        }
        function handleEscape(e: KeyboardEvent) {
          if (e.key === 'Escape') setOpen(false);
        }
        document.addEventListener('mousedown', handleClickOutside);
        document.addEventListener('keydown', handleEscape);
        return () => {
          document.removeEventListener('mousedown', handleClickOutside);
          document.removeEventListener('keydown', handleEscape);
        };
      }, [open]);

      const avatar = user.image ? (
        <Image
          src={user.image}
          alt={user.name ?? '用户头像'}
          width={32}
          height={32}
          style={{ borderRadius: '50%', flexShrink: 0 }}
        />
      ) : (
        // Monogram fallback — UI-SPEC §UserChip monogram
        <span
          role="img"
          aria-label={initialOf(user.name, user.email)}
          style={{
            display: 'inline-flex',
            alignItems: 'center',
            justifyContent: 'center',
            width: 32,
            height: 32,
            borderRadius: '50%',
            background: 'var(--accent-100)',
            color: 'var(--accent-700)',
            fontWeight: 600,
            fontSize: 14,
            flexShrink: 0,
          }}
        >
          {initialOf(user.name, user.email)}
        </span>
      );

      return (
        <div ref={containerRef} style={{ margin: '0 12px 12px', position: 'relative' }}>
          <button
            type="button"
            aria-haspopup="menu"
            aria-expanded={open}
            onClick={() => setOpen((v) => !v)}
            style={{
              display: 'flex',
              alignItems: 'center',
              gap: 8,
              width: '100%',
              padding: '6px 8px',
              borderRadius: 8,
              background: 'transparent',
              border: '1px solid var(--line-weak)',
              cursor: 'pointer',
            }}
          >
            {avatar}
            <span style={{ flex: 1, textAlign: 'left', fontSize: 14, color: 'var(--ink-900)' }}>{displayName}</span>
            <Icon name={open ? 'chevron-up' : 'chevron-down'} size={16} />
          </button>

          {open && (
            <div
              role="menu"
              style={{
                position: 'absolute',
                bottom: '100%',
                left: 0,
                right: 0,
                marginBottom: 4,
                background: 'var(--surface-0)',
                border: '1px solid var(--line)',
                borderRadius: 8,
                boxShadow: 'var(--shadow-md)',
                minWidth: 160,
              }}
            >
              <form action={signOutAction}>
                <button
                  type="submit"
                  role="menuitem"
                  style={{
                    display: 'flex',
                    alignItems: 'center',
                    gap: 8,
                    width: '100%',
                    minHeight: 40,
                    padding: '0 12px',
                    background: 'transparent',
                    border: 0,
                    fontSize: 14,
                    color: 'var(--ink-900)',
                    cursor: 'pointer',
                    textAlign: 'left',
                  }}
                >
                  <Icon name="log-out" size={16} />
                  退出登录
                </button>
              </form>
            </div>
          )}
        </div>
      );
    }
    ```

    Note on chevron-up/down: if the existing IconName union includes both, use them; otherwise use `chevron-down` only.

    Fill in `tests/unit/user-chip.test.tsx` and `tests/unit/user-chip-signout.test.tsx` with Testing Library assertions covering all 5 Behavior tests. Mock next/image as needed.
