    ```typescript
    /**
     * Auth.js v5 configuration — Phase 5 AUTH-01..08, D-01..D-21.

     * Database session strategy (D-03) with two-layer ban enforcement (D-05 Layer 1).

     *
     * Consumed by:
     *   - src/lib/auth/index.ts (NextAuth(authConfig) → handlers/auth/signIn/signOut)
     *   - src/app/api/auth/[...nextauth]/route.ts (handlers.GET/POST)
     *   - tests/unit/auth-config.test.ts (shape assertions)
     */
    import type { NextAuthConfig } from 'next-auth';
    import { DrizzleAdapter } from '@auth/drizzle-adapter';
    import { eq } from 'drizzle-orm';
    import { db } from '@/lib/db/client';
    import { users, accounts, sessions, verificationTokens } from '@/lib/db/schema';

    export const authConfig = {
      adapter: DrizzleAdapter(db, {
        usersTable: users,
        accountsTable: accounts,
        sessionsTable: sessions,
        verificationTokensTable: verificationTokens,
      }),
      session: { strategy: 'database' },
      redirectProxyUrl: process.env.AUTH_REDIRECT_PROXY_URL,
      providers: [], // Plan 03 populates GitHub + Resend + Google
      callbacks: {
        async session({ session, user }) {
          // D-05 Layer 1 — ban check
          if ((user as any).isBanned) {
            return null as any;
          }
          // D-08 — expose id + role; mirror image from users.image || avatar_url
          session.user.id = user.id;
          (session.user as any).role = (user as any).role;
          session.user.image = (user as any).image ?? (user as any).avatarUrl ?? null;
          return session;
        },
      },
      events: {
        async signIn({ user }) {
          // D-09 — touch last_seen_at only on sign-in
          if (user.id) {
            await db.update(users).set({ lastSeenAt: new Date() }).where(eq(users.id, user.id));
          }
        },
        async linkAccount({ user, profile }) {
          // D-04 — mirror image to avatar_url for Phase 4 code reading avatar_url
          const image = (profile as any)?.avatar_url ?? (profile as any)?.picture;
          if (image && user.id) {
            await db.update(users).set({ avatarUrl: image }).where(eq(users.id, user.id));
          }
        },
      },
    } satisfies NextAuthConfig;
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → /api/auth/* | Public endpoint accepting OAuth callbacks + magic-link tokens |
| /api/auth → Neon DB | Session writes via adapter |
| session callback → users.is_banned | Security-critical check; Layer 1 of two-layer ban enforcement |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-5-01 | Tampering | @auth/drizzle-adapter queries | mitigate | Adapter uses Drizzle parameterized queries only — no string interpolation on user-controlled input |
| T-5-02 | Spoofing | /api/auth OAuth state cookie | mitigate | Auth.js v5 built-in state + PKCE defaults via redirectProxyUrl config (RESEARCH Pattern 3 A2) |
| T-5-03 | Elevation of privilege | callbacks.session ban bypass | mitigate | Returns null when user.isBanned=true; tested via integration test in Task 2 |
| T-5-09 | Tampering | callbackUrl after signIn | mitigate | Auth.js v5 built-in same-origin callbackUrl validation |
| T-5-10 | Tampering | preview OAuth session cookie | mitigate | AUTH_SECRET documented as byte-identical across scopes in .env.example comment + RESEARCH §Pattern 3 |
| T-5-11 | PIPL data residency | user PII | mitigate | Auth.js self-hosted via @auth/drizzle-adapter on Neon — no third-party PII vendor (no Clerk) |
</threat_model>
