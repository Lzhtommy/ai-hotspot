    ```typescript
    /**
     * Chinese magic-link email sender — Phase 5 AUTH-03, D-07.
     *
     * Called by Auth.js Resend provider's sendVerificationRequest hook.
     * POSTs Chinese-body email to Resend HTTP API (verifies 10-minute TTL per
     * Auth.js default; UI-SPEC §Email body for exact copy).
     *
     * Consumed by:
     *   - src/lib/auth/config.ts (Resend provider)
     *   - tests/unit/provider-resend.test.ts (shape + body assertions)
     */

    export interface SendMagicLinkDeps {
      fetch?: typeof globalThis.fetch;
    }

    export interface SendMagicLinkParams {
      identifier: string;
      url: string;
      provider: { apiKey?: string; from?: string };
      expires: Date;
    }

    export class MagicLinkError extends Error {
      constructor(message: string, public status?: number) {
        super(message);
        this.name = 'MagicLinkError';
      }
    }

    const SUBJECT = 'AI Hotspot 登录链接';

    function buildTextBody(url: string): string {
      return [
        '你好，',
        '',
        '请点击以下链接登录 AI Hotspot：',
        '',
        url,
        '',
        '链接 10 分钟内有效。如果你没有申请登录，请忽略此邮件。',
        '',
        'AI Hotspot 团队',
      ].join('\n');
    }

    function buildHtmlBody(url: string): string {
      return `<!DOCTYPE html><html><body style="font-family:-apple-system,Helvetica,Arial,sans-serif;line-height:1.6;color:#3A3833;padding:24px;">
<p>你好，</p>
<p>请点击以下链接登录 AI Hotspot：</p>
<p><a href="${url}" style="display:inline-block;padding:10px 20px;background:#D4911C;color:#fff;text-decoration:none;border-radius:4px;">登录 AI Hotspot</a></p>
<p style="color:#766F5B;font-size:14px;">链接 10 分钟内有效。如果你没有申请登录，请忽略此邮件。</p>
<p style="color:#766F5B;font-size:14px;">— AI Hotspot 团队</p>
</body></html>`;
    }

    export async function sendChineseMagicLink(
      params: SendMagicLinkParams,
      deps?: SendMagicLinkDeps,
    ): Promise<void> {
      const fetchFn = deps?.fetch ?? globalThis.fetch;
      const { identifier, url, provider } = params;
      if (!provider.apiKey) throw new MagicLinkError('Resend apiKey missing');
      if (!provider.from) throw new MagicLinkError('Resend from missing');

      const res = await fetchFn('https://api.resend.com/emails', {
        method: 'POST',
        headers: {
          Authorization: `Bearer ${provider.apiKey}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          from: provider.from,
          to: identifier,
          subject: SUBJECT,
          text: buildTextBody(url),
          html: buildHtmlBody(url),
        }),
      });

      if (!res.ok) {
        const detail = await res.text().catch(() => '');
        throw new MagicLinkError(`Resend send failed: ${res.status}`, res.status);
      }
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| /api/auth/callback/github|google → Auth.js | OAuth provider redirects with code; state verified via HMAC |
| /api/auth/signin/email → Resend API | Magic-link token written to verification_tokens; email POSTed to external SaaS |
| next/image → avatar CDN | Public image fetch, no auth bypass |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-5-02 | Spoofing | OAuth state/PKCE | mitigate | Auth.js v5 default state + PKCE checks on GitHub + Google providers (RESEARCH §Don't Hand-Roll) |
| T-5-04 | Elevation of privilege | GitHub profile spoofing → admin | mitigate | profile() function does NOT set role; role column stays default 'user'. Admin promotion is out-of-band SQL (Plan 10 runbook) |
| T-5-05 | Replay | Magic-link token replay | mitigate | verification_tokens composite PK (Plan 01) + Auth.js single-use adapter semantics + 10-minute TTL |
| T-5-07 | XSS | user-set name/image | mitigate | React escapes name on render; next/image only accepts URLs matching remotePatterns allowlist (this plan) |
| T-5-12 | SSRF | next/image server-side fetch | mitigate | images.remotePatterns explicit allowlist: only avatars.githubusercontent.com + lh3.googleusercontent.com. No wildcard. |
</threat_model>
