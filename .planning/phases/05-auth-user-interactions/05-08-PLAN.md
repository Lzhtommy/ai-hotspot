    ```typescript
    import { redirect } from 'next/navigation';
    import { desc, eq, and } from 'drizzle-orm';
    import { auth } from '@/lib/auth';
    import { db } from '@/lib/db/client';
    import { favorites, items, sources, clusters, votes } from '@/lib/db/schema';
    import { FeedTopBar } from '@/components/feed/feed-top-bar';
    import { Timeline } from '@/components/feed/timeline';
    import { FavoritesEmpty } from './favorites-empty';

    export const dynamic = 'force-dynamic';

    export default async function FavoritesPage() {
      const session = await auth();
      if (!session?.user?.id) {
        redirect('/');
      }
      const userId = session.user.id;

      // Mirror the SELECT shape from src/lib/feed/get-feed.ts FeedListItem
      const rows = await db
        .select({
          // ... full FeedListItem shape: id, title, summary, hotnessScore, recommendationReason,
          //     publishedAt, sourceId, sourceName, sourceBadgeUrl, clusterId, clusterMemberCount,
          //     tags, imageUrl, externalUrl, etc.
          id: items.id,
          title: items.title,
          summary: items.summary,
          hotnessScore: items.hotnessScore,
          recommendationReason: items.recommendationReason,
          publishedAt: items.publishedAt,
          status: items.status,
          externalUrl: items.externalUrl,
          sourceId: items.sourceId,
          sourceName: sources.name,
          clusterId: items.clusterId,
          clusterMemberCount: clusters.memberCount,
          favoritedAt: favorites.createdAt,
        })
        .from(favorites)
        .innerJoin(items, eq(items.id, favorites.itemId))
        .leftJoin(sources, eq(sources.id, items.sourceId))
        .leftJoin(clusters, eq(clusters.id, items.clusterId))
        .where(and(eq(favorites.userId, userId), eq(items.status, 'published')))
        .orderBy(desc(favorites.createdAt));

      // Fetch user's vote state for the same items (to thread into FeedCardActions)
      const voteRows = rows.length
        ? await db
            .select({ itemId: votes.itemId, value: votes.value })
            .from(votes)
            .where(eq(votes.userId, userId))
        : [];
      const voteMap = new Map(voteRows.map(v => [v.itemId.toString(), v.value]));
      const favoriteMap = new Map(rows.map(r => [r.id.toString(), true]));

      const itemsForDisplay = rows.map(r => ({
        ...r,
        interaction: {
          favorited: favoriteMap.get(r.id.toString()) ?? false,
          vote: (voteMap.get(r.id.toString()) ?? 0) as -1 | 0 | 1,
        },
      }));

      return (
        <>
          <FeedTopBar
            view="favorites"
            pathname="/favorites"
            subtitle={rows.length > 0 ? `共 ${rows.length} 条` : '还没有收藏'}
          />
          {rows.length === 0 ? (
            <FavoritesEmpty authenticated />
          ) : (
            <Timeline items={itemsForDisplay} isAuthenticated={true} />
          )}
        </>
      );
    }
    ```

If Timeline doesn't accept `isAuthenticated` prop yet, either add the prop plumbing here (pass-through to FeedCard → FeedCardActions) or rely on Plan 07 Task 3 which may have established this thread.
