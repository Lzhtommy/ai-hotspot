    /**
     * Auth guard — Phase 5 AUTH-01, D-05 Layer 2.
     *
     * Pure function: caller injects session (from auth()) + db. Tests mock both.
     * Throws AuthError with code 'UNAUTHENTICATED' (no session) or 'FORBIDDEN' (banned).
     */
    import { eq } from 'drizzle-orm';
    import { db as realDb } from '@/lib/db/client';
    import { users } from '@/lib/db/schema';
    import type { Session } from 'next-auth';

    export class AuthError extends Error {
      constructor(public code: 'UNAUTHENTICATED' | 'FORBIDDEN', message?: string) {
        super(message ?? code);
        this.name = 'AuthError';
      }
    }

    export interface RequireLiveUserDeps {
      db?: typeof realDb;
    }

    export async function requireLiveUserCore(
      session: Session | null,
      deps?: RequireLiveUserDeps,
    ): Promise<string> {
      if (!session?.user?.id) throw new AuthError('UNAUTHENTICATED');
      const dbx = deps?.db ?? realDb;
      const [row] = await dbx
        .select({ isBanned: users.isBanned })
        .from(users)
        .where(eq(users.id, session.user.id));
      if (!row || row.isBanned) throw new AuthError('FORBIDDEN');
      return session.user.id;
    }
    ```

    /**
     * Votes core — Phase 5 VOTE-01, VOTE-02, VOTE-04, D-12.
     *
     * Exclusive toggle state machine:
     *   no row      + value=v  → insert(value=v)         → { vote: v }
     *   row value=v + value=v  → delete                  → { vote: 0 }
     *   row value=v + value=-v → update(value=-v)        → { vote: -v }
     *
     * Input value validated to {-1, +1}; other values throw.
     */
    import { and, eq } from 'drizzle-orm';
    import { db as realDb } from '@/lib/db/client';
    import { votes } from '@/lib/db/schema';

    export type VoteValue = -1 | 1;
    export type VoteState = -1 | 0 | 1;

    export interface VoteDeps {
      db?: typeof realDb;
    }

    export class VoteValueError extends Error {
      constructor() {
        super('VoteValueError: value must be -1 or +1');
        this.name = 'VoteValueError';
      }
    }

    export async function voteItemCore(
      params: { userId: string; itemId: bigint; value: VoteValue },
      deps?: VoteDeps,
    ): Promise<{ vote: VoteState }> {
      const { userId, itemId, value } = params;
      if (value !== 1 && value !== -1) throw new VoteValueError();

      const dbx = deps?.db ?? realDb;
      const [existing] = await dbx
        .select({ value: votes.value })
        .from(votes)
        .where(and(eq(votes.userId, userId), eq(votes.itemId, itemId)));

      if (!existing) {
        await dbx.insert(votes).values({ userId, itemId, value });
        return { vote: value };
      }

      if (existing.value === value) {
        await dbx
          .delete(votes)
          .where(and(eq(votes.userId, userId), eq(votes.itemId, itemId)));
        return { vote: 0 };
      }

      await dbx
        .update(votes)
        .set({ value })
        .where(and(eq(votes.userId, userId), eq(votes.itemId, itemId)));
      return { vote: value };
    }
    ```
