    ```tsx
    'use client';

    /**
     * FeedCardActions — Phase 5 FAV-01, FAV-02, VOTE-01, VOTE-02, VOTE-03, VOTE-04, D-10..D-14.
     *
     * Three action buttons (star / check / x) with useOptimistic. Anonymous clicks
     * dispatch open-login-modal (Phase 4 seam preserved). Authenticated clicks call
     * server actions and optimistically update UI.
     */

    import { useOptimistic, useState, startTransition } from 'react';
    import { IconButton } from '@/components/layout/icon-button';
    import { favoriteItem, unfavoriteItem } from '@/server/actions/favorites';
    import { voteItem } from '@/server/actions/votes';

    // VOTE-03 honest-copy (UI-SPEC §Copywriting Contract — lives as a file-scope constant)
    const PERSONALIZATION_COPY = '个性化推荐即将上线';
    const ROLLBACK_ERROR_COPY = '操作失败，请重试。';

    type Interaction = { favorited: boolean; vote: -1 | 0 | 1 };

    interface FeedCardActionsProps {
      itemId: string;
      url: string;
      sourceUrl?: string | null;
      initial: Interaction;
      isAuthenticated: boolean;
    }

    function domainOf(url: string): string {
      try { return new URL(url).hostname.replace(/^www\./, ''); } catch { return url; }
    }

    function openLoginModal() {
      document.dispatchEvent(new CustomEvent('open-login-modal'));
    }

    export function FeedCardActions({ itemId, url, sourceUrl, initial, isAuthenticated }: FeedCardActionsProps) {
      const [optimistic, setOptimistic] = useOptimistic<Interaction, Partial<Interaction>>(
        initial,
        (curr, patch) => ({ ...curr, ...patch }),
      );
      const [errorShown, setErrorShown] = useState(false);

      function showErrorBriefly() {
        setErrorShown(true);
        setTimeout(() => setErrorShown(false), 3000);
      }

      function handleFavorite() {
        if (!isAuthenticated) { openLoginModal(); return; }
        const next = !optimistic.favorited;
        startTransition(async () => {
          setOptimistic({ favorited: next });
          try {
            const res = next ? await favoriteItem(itemId) : await unfavoriteItem(itemId);
            startTransition(() => setOptimistic({ favorited: res.favorited }));
          } catch {
            startTransition(() => setOptimistic({ favorited: !next }));
            showErrorBriefly();
          }
        });
      }

      function handleVote(desired: 1 | -1) {
        if (!isAuthenticated) { openLoginModal(); return; }
        const currentVote = optimistic.vote;
        const next: -1 | 0 | 1 = currentVote === desired ? 0 : desired;
        startTransition(async () => {
          setOptimistic({ vote: next });
          try {
            const res = await voteItem(itemId, desired);
            startTransition(() => setOptimistic({ vote: res.vote }));
          } catch {
            startTransition(() => setOptimistic({ vote: currentVote }));
            showErrorBriefly();
          }
        });
      }

      return (
        <div style={{ marginTop: 14, paddingTop: 12, borderTop: '1px solid var(--line-weak)' }}>
          <div style={{ display: 'flex', alignItems: 'center', gap: 4 }}>
            <IconButton
              icon="star"
              tone="accent"
              active={optimistic.favorited}
              aria-pressed={optimistic.favorited}
              aria-label={optimistic.favorited ? '已收藏' : '收藏'}
              title={optimistic.favorited ? '已收藏' : '收藏'}
              onClick={handleFavorite}
            />
            <IconButton
              icon="check"
              tone="success"
              active={optimistic.vote === 1}
              aria-pressed={optimistic.vote === 1}
              aria-label={optimistic.vote === 1 ? '已点赞' : '点赞'}
              title={optimistic.vote === 1 ? '已点赞' : '点赞'}
              onClick={() => handleVote(1)}
            />
            <IconButton
              icon="x"
              tone="danger"
              active={optimistic.vote === -1}
              aria-pressed={optimistic.vote === -1}
              aria-label={optimistic.vote === -1 ? '已点踩' : '点踩'}
              title={optimistic.vote === -1 ? '已点踩' : '点踩'}
              onClick={() => handleVote(-1)}
            />
            <a
              href={url}
              target="_blank"
              rel="noopener noreferrer"
              style={{ marginLeft: 'auto', fontSize: 12, color: 'var(--ink-500)', textDecoration: 'none' }}
            >
              {domainOf(url)} ↗
            </a>
          </div>

          {/* VOTE-03 honest copy / rollback error */}
          {errorShown ? (
            <p role="alert" style={{ marginTop: 8, fontSize: 12, color: 'var(--danger-500)' }}>
              {ROLLBACK_ERROR_COPY}
            </p>
          ) : (
            <p aria-live="off" style={{ marginTop: 8, fontSize: 12, color: 'var(--ink-500)' }}>
              {PERSONALIZATION_COPY}
            </p>
          )}
        </div>
      );
    }
    ```
