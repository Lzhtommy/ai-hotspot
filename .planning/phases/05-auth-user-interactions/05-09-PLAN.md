    ```typescript
    import { randomUUID } from 'node:crypto';
    import { eq } from 'drizzle-orm';
    import { makeTestDb } from './db';
    import * as schema from '@/lib/db/schema';

    export interface SeededSession {
      sessionToken: string;
      userId: string;
      email: string;
      cookie: {
        name: string;
        value: string;
        domain: string;
        path: string;
        expires: number;
        httpOnly: boolean;
        secure: boolean;
        sameSite: 'Lax' | 'Strict' | 'None';
      };
      cleanup: () => Promise<void>;
    }

    export async function seedSession(options: {
      email?: string;
      name?: string;
      isBanned?: boolean;
      role?: string;
      baseUrl?: string;
    } = {}): Promise<SeededSession> {
      const db = makeTestDb();
      const email = options.email ?? `e2e-${randomUUID().slice(0, 8)}@test.local`;
      const [userRow] = await db
        .insert(schema.users)
        .values({
          email,
          name: options.name ?? 'E2E User',
          role: options.role ?? 'user',
          isBanned: options.isBanned ?? false,
        })
        .returning({ id: schema.users.id });
      const userId = userRow.id;

      const sessionToken = randomUUID();
      const expires = new Date(Date.now() + 30 * 24 * 60 * 60 * 1000);
      await db.insert(schema.sessions).values({ sessionToken, userId, expires });

      const baseUrl = options.baseUrl ?? 'http://localhost:3000';
      const host = new URL(baseUrl).hostname;
      const isSecure = baseUrl.startsWith('https://');

      return {
        sessionToken,
        userId,
        email,
        cookie: {
          name: isSecure ? '__Secure-authjs.session-token' : 'authjs.session-token',
          value: sessionToken,
          domain: host,
          path: '/',
          expires: Math.floor(expires.getTime() / 1000),
          httpOnly: true,
          secure: isSecure,
          sameSite: 'Lax',
        },
        cleanup: async () => {
          await db.delete(schema.sessions).where(eq(schema.sessions.sessionToken, sessionToken));
          await db.delete(schema.users).where(eq(schema.users.id, userId));
        },
      };
    }
    ```

    This enables E2E specs to call `await seedSession()` and then pass the cookie to Playwright's `context.addCookies([seeded.cookie])`.

    import { test, expect } from '@playwright/test';
    import { seedSession } from '../helpers/seed-session';

    test.describe('AUTH-02 GitHub OAuth (post-auth state simulation)', () => {
      test('seeded session is recognized and UserChip shows authenticated state', async ({ page, context }) => {
        const seeded = await seedSession({ name: 'GitHub User', email: 'gh-user@test.local' });
        try {
          await context.addCookies([seeded.cookie]);
          await page.goto('/');
          // UserChip authenticated branch should render name
          await expect(page.getByRole('button', { name: /GitHub User|gh-user/ })).toBeVisible();
          // Anonymous 登录 ghost chip should NOT be visible
          await expect(page.getByRole('button', { name: '登录' })).toHaveCount(0);
        } finally {
          await seeded.cleanup();
        }
      });

      test('clicking 退出登录 clears session and returns to anonymous', async ({ page, context }) => {
        const seeded = await seedSession();
        try {
          await context.addCookies([seeded.cookie]);
          await page.goto('/');
          await page.getByRole('button', { name: new RegExp(seeded.email.split('@')[0]) }).click();
          await page.getByRole('menuitem', { name: '退出登录' }).click();
          // After signOut, UserChip reverts to anonymous
          await expect(page.getByRole('button', { name: '登录' })).toBeVisible({ timeout: 5000 });

          await seeded.cleanup();
        }

    test.describe('VOTE-04 + FAV-01 anon → sign-in → favorite', () => {
      test('anonymous star click opens modal; after seeded sign-in, star click persists', async ({ page, context }) => {
        const db = makeTestDb();
        await page.goto('/');
        // Find the first feed card star button
        const starButtons = page.getByRole('button', { name: /^(收藏|已收藏)$/ });
        const firstStar = starButtons.first();

        await firstStar.click();
        // Modal opens (login-modal-heading from Phase 4)
        await expect(page.locator('[id="login-modal-heading"]')).toBeVisible({ timeout: 5000 });
        // Dismiss for the simulated "seeded sign-in"
        await page.getByRole('button', { name: '稍后再说' }).click();

        const seeded = await seedSession();
        try {
          await context.addCookies([seeded.cookie]);
          await page.reload();

          // Click star again — should persist
          const starBtn = page.getByRole('button', { name: /^(收藏|已收藏)$/ }).first();
          await starBtn.click();

          // Assert persist via DB (optimistic UI may say 已收藏; DB is the truth)
          // Re-fetch any favorite row for this user
          const rows = await db.select().from(favorites).where(eq(favorites.userId, seeded.userId));
          expect(rows.length).toBeGreaterThanOrEqual(1);

          // Cleanup favorites
          await db.delete(favorites).where(eq(favorites.userId, seeded.userId));
        } finally {
          await seeded.cleanup();
        }
      });
