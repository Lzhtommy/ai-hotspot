      ```tsx
      'use client';
      import { useState } from 'react';
      import { useQueryState, parseAsArrayOf, parseAsString } from 'nuqs';
      import { Button } from '@/components/layout/button';
      import { Tag } from '@/components/layout/tag';

      export function FilterPopover({ availableTags, availableSources }: {
        availableTags: string[];
        availableSources: Array<{ id: number; name: string }>;
      }) {
        const [open, setOpen] = useState(false);
        const [tags, setTags] = useQueryState('tags', parseAsArrayOf(parseAsString).withDefault([]).withOptions({ shallow: false }));
        const [source, setSource] = useQueryState('source', parseAsString.withDefault('').withOptions({ shallow: false }));
        const toggleTag = (t: string) => setTags(tags.includes(t) ? tags.filter(x => x !== t) : [...tags, t]);
        const clear = () => { setTags([]); setSource(''); };
        return (
          <div className="relative inline-block">
            <Button variant="ghost" size="md" onClick={() => setOpen(v => !v)} aria-expanded={open} aria-controls="filter-popover">过滤</Button>
            {open && (
              <div id="filter-popover" role="dialog" className="absolute right-0 mt-[8px] z-20 w-[320px] rounded-[8px] border border-[var(--line)] bg-[var(--surface-0)] shadow-md p-[16px]">
                <div className="flex items-center justify-between">
                  <h3 className="text-[14px] font-semibold text-[var(--ink-900)]">筛选</h3>
                  <button onClick={clear} className="text-[12px] text-[var(--fg-3)] hover:text-[var(--ink-900)]">清除</button>
                </div>
                <div className="mt-[12px]">
                  <div className="text-[10.5px] uppercase tracking-[0.08em] text-[var(--fg-3)] mb-[8px]">标签</div>
                  <div className="flex flex-wrap gap-[5px]">
                    {availableTags.map(t => (
                      <button key={t} type="button" aria-pressed={tags.includes(t)} onClick={() => toggleTag(t)}>
                        <Tag label={t} active={tags.includes(t)} />
                      </button>
                    ))}
                  </div>
                </div>
                <div className="mt-[16px]">
                  <div className="text-[10.5px] uppercase tracking-[0.08em] text-[var(--fg-3)] mb-[8px]">信源</div>
                  <div className="flex flex-wrap gap-[5px]">
                    <button type="button" aria-pressed={source === ''} onClick={() => setSource('')}>
                      <Tag label="全部信源" active={source === ''} />
                    </button>
                    {availableSources.map(s => (
                      <button key={s.id} type="button" aria-pressed={source === String(s.id)} onClick={() => setSource(String(s.id))}>
                        <Tag label={s.name} active={source === String(s.id)} />
                      </button>
                    ))}
                  </div>
                </div>
              </div>
            )}
          </div>
        );
      }
      ```
